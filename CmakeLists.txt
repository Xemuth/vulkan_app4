cmake_minimum_required(VERSION 3.10)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

find_package(Vulkan)

project(vulkan_app3 C CXX)

include("${CMAKE_CURRENT_SOURCE_DIR}/third_party/CmakeLists.txt")

set(ENABLE_VALIDATION_LAYER CACHE BOOL OFF)
set(ENABLE_LOGGER CACHE BOOL OFF)

file(GLOB_RECURSE vulkan_app3_src
    CONFIGURE_DEPENDS 
    "src/*.h"
    "src/*.cpp"
    "src/*.c"
    "src/*.hpp"
    "include/*.h"
    "include/*.cpp"
    "include/*.c"
    "include/*.hpp"
)

add_executable(vulkan_app3 ${vulkan_app3_src})

set_target_properties(vulkan_app3
    PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/../bin/${CMAKE_BUILD_TYPE}"
)

target_compile_definitions(vulkan_app3 PUBLIC
    GLFW_INCLUDE_VULKAN
)

if(${ENABLE_VALIDATION_LAYER})
    add_compile_definitions(vulkan_app3 PUBLIC -DENABLE_VALIDATION_LAYER)
endif()

if(${ENABLE_LOGGER})
    add_compile_definitions(vulkan_app3 PUBLIC -DENABLE_LOGGER)
endif()

target_compile_options(vulkan_app3 PUBLIC -nostdlib)

target_include_directories(vulkan_app3 PUBLIC
    "${CMAKE_CURRENT_SOURCE_DIR}/include"
    ${Vulkan_INCLUDE_DIRS}
    glfw
    spdlog
    ${glm_include_path}
)

target_link_libraries(vulkan_app3 PUBLIC 
    glfw
    spdlog
)
